// Chương trình Java giải hệ phương trình 2 ẩn bằng công thức Cramer
import java.util.Scanner;

public class HePhuongTrinh2An {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Nhập các hệ số
        System.out.print("Nhap a: ");
        double a = sc.nextDouble();
        System.out.print("Nhap b: ");
        double b = sc.nextDouble();
        System.out.print("Nhap c: ");
        double c = sc.nextDouble();

        System.out.print("Nhap d: ");
        double d = sc.nextDouble();
        System.out.print("Nhap e: ");
        double e = sc.nextDouble();
        System.out.print("Nhap f: ");
        double f = sc.nextDouble();

        // Tính định thức
        double D = a * e - b * d;
        double Dx = c * e - b * f;
        double Dy = a * f - c * d;

        System.out.println("D = " + D);
        System.out.println("Dx = " + Dx);
        System.out.println("Dy = " + Dy);

        // Kiểm tra và xuất kết quả
        if (D != 0) {
            double x = Dx / D;
            double y = Dy / D;
            System.out.println("He co nghiem duy nhat:");
            System.out.println("x = " + x);
            System.out.println("y = " + y);
        } else {
            if (Dx == 0 && Dy == 0) {
                System.out.println("He phuong trinh co vo so nghiem.");
            } else {
                System.out.println("He phuong trinh vo nghiem.");
            }
        }

        sc.close();
    }
}
